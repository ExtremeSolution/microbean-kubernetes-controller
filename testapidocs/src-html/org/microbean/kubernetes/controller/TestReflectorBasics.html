056public class TestReflectorBasics {
057
058  public TestReflectorBasics() {
059    super();
060  }
061
062  @Test
063  public void testBasics() throws Exception {
064    assumeFalse(Boolean.getBoolean("skipClusterTests"));
065
066    // We'll use this as our "known objects".
067    final Map<Object, ConfigMap> configMaps = new HashMap<>();
068
069    // Create a new EventCache implementation that "knows about" our
070    // known objects.
071    final EventQueueCollection<ConfigMap> eventQueues = new EventQueueCollection<>(configMaps, 16, 0.75f);
072
073    // Create a consumer that can remove and process EventQueues from
074    // our EventCache implementation.  It will also update our "known
075    // objects" as necessary.
076    final Consumer<? super EventQueue<? extends ConfigMap>> siphon =
077      new ResourceTrackingEventQueueConsumer<ConfigMap>(configMaps) {
078        @Override
079        protected final void accept(final AbstractEvent<? extends ConfigMap> event) {
080          assertNotNull(event);
081          System.out.println("*** received event: " + event);
082        }
083      };
084
085    // Begin sucking EventQueue instances out of the cache on a
086    // separate Thread.  Obviously there aren't any yet.  This creates
087    // a new (daemon) Thread and starts it.  It will block
088    // immediately, waiting for new EventQueues to show up in our
089    // EventQueueCollection.
090    eventQueues.start(siphon);
091
092    // Connect to Kubernetes using a combination of system properties,
093    // environment variables and ~/.kube/config settings as detailed
094    // here:
095    // https://github.com/fabric8io/kubernetes-client/blob/v3.2.0/README.md#configuring-the-client.
096    // We'll use this client when we create a Reflector below.
097    final DefaultKubernetesClient client = new DefaultKubernetesClient();
098
099    // Now create a Reflector that we'll then hook up to Kubernetes
100    // and instruct to "reflect" its events "into" our
101    // EventQueueCollection, thus making EventQueues available to the
102    // Consumer we built above.
103    final Reflector<ConfigMap> reflector =
104      new Reflector<ConfigMap>(client.configMaps(),
105                               eventQueues,
106                               Duration.ofSeconds(10));
107
108    // Start the reflection process: this effectively puts EventQueue
109    // instances into the cache.  This creates a new (daemon) Thread
110    // and starts it.
111    System.out.println("*** starting reflector");
112    reflector.start();
113
114    // Sleep for a bit on the main thread so you can see what's going
115    // on and try adding some resources to Kubernetes in a terminal
116    // window.  Watch as the consumer we built above will report on
117    // all the additions, updates, deletions and synchronizations.
118    Thread.sleep(1L * 60L * 1000L);
119
120    // Close the Reflector.  This cancels any scheduled
121    // synchronization tasks.
122    System.out.println("*** closing reflector");
123    reflector.close();
124    
125    // Close the client, now that no one will be calling it anymore.
126    System.out.println("*** closing client");
127    client.close();
128
129    // Shut down reception of events now that no one is making any
130    // more of them.
131    System.out.println("*** closing eventQueues");
132    eventQueues.close();
133  }
