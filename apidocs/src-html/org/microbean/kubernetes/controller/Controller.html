017package org.microbean.kubernetes.controller;
018
019import java.io.Closeable;
020import java.io.IOException;
021
022import java.time.Duration;
023
024import java.util.Map;
025import java.util.Objects;
026
027import java.util.concurrent.ScheduledExecutorService;
028
029import java.util.function.Consumer;
030
031import io.fabric8.kubernetes.api.model.HasMetadata;
032import io.fabric8.kubernetes.api.model.KubernetesResourceList;
033
034import io.fabric8.kubernetes.client.Watcher;
035
036import io.fabric8.kubernetes.client.dsl.Listable;
037import io.fabric8.kubernetes.client.dsl.VersionWatchable;
038
039import net.jcip.annotations.Immutable;
040
041@Immutable
042public class Controller<T extends HasMetadata> {
043
044  private final Reflector<T> reflector;
045
046  private final EventQueueCollection<T> eventCache;
047
048  private final Consumer<? super EventQueue<? extends T>> siphon;
049
050  @SuppressWarnings("rawtypes")
051  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Controller(final X operation,
052                                                                                                                               final Consumer<? super EventQueue<? extends T>> siphon) {
053    this(operation, null, null, null, siphon);
054  }
055  
056  @SuppressWarnings("rawtypes")
057  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Controller(final X operation,
058                                                                                                                               final Map<Object, T> knownObjects,
059                                                                                                                               final Consumer<? super EventQueue<? extends T>> siphon) {
060    this(operation, null, null, knownObjects, siphon);
061  }
062  
063  @SuppressWarnings("rawtypes")
064  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Controller(final X operation,
065                                                                                                                               final Duration synchronizationInterval,
066                                                                                                                               final Consumer<? super EventQueue<? extends T>> siphon) {
067    this(operation, null, synchronizationInterval, null, siphon);
068  }
069
070  @SuppressWarnings("rawtypes")
071  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Controller(final X operation,
072                                                                                                                               final Duration synchronizationInterval,
073                                                                                                                               final Map<Object, T> knownObjects,
074                                                                                                                               final Consumer<? super EventQueue<? extends T>> siphon) {
075    this(operation, null, synchronizationInterval, knownObjects, siphon);
076  }
077
078  @SuppressWarnings("rawtypes")
079  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Controller(final X operation,
080                                                                                                                               final ScheduledExecutorService synchronizationExecutorService,
081                                                                                                                               final Duration synchronizationInterval,
082                                                                                                                               final Map<Object, T> knownObjects,
083                                                                                                                               final Consumer<? super EventQueue<? extends T>> siphon) {
084    super();
085    this.siphon = Objects.requireNonNull(siphon);
086    this.eventCache = new ControllerEventQueueCollection(knownObjects);
087    this.reflector = new ControllerReflector(operation, synchronizationExecutorService, synchronizationInterval);
088  }
089
090  public final void start() {
091    this.eventCache.start(siphon);
092    this.reflector.start();
093  }
094
095  public final void close() throws IOException {
096    Exception throwMe = null;
097    try {
098      this.reflector.close();
099    } catch (final Exception everything) {
100      throwMe = everything;
101    }
102
103    try {
104      this.eventCache.close();
105    } catch (final RuntimeException runtimeException) {
106      if (throwMe == null) {
107        throw runtimeException;
108      }
109      assert throwMe instanceof IOException;
110      throwMe.addSuppressed(runtimeException);
111      throw (IOException)throwMe;
112    }
113  }
114
115  protected boolean shouldSynchronize() {
116    return true;
117  }
118
119  protected void onClose() {
120
121  }
122
123  protected Object getKey(final T resource) {
124    return HasMetadatas.getKey(resource);
125  }
126
127  protected Event<T> createEvent(final Object source, final Event.Type eventType, final T resource) {
128    return new Event<T>(source, eventType, null, resource);
129  }
130
131  protected EventQueue<T> createEventQueue(final Object key) {
132    return new EventQueue<T>(key);
133  }
134
135  private final class ControllerEventQueueCollection extends EventQueueCollection<T> {
136
137    private ControllerEventQueueCollection(final Map<?, ? extends T> knownObjects) {
138      super(knownObjects);
139    }
140    
141    @Override
142    protected final Event<T> createEvent(final Object source, final Event.Type eventType, final T resource) {
143      return Controller.this.createEvent(source, eventType, resource);
144    }
145    
146    @Override
147    protected final EventQueue<T> createEventQueue(final Object key) {
148      return Controller.this.createEventQueue(key);
149    }
150    
151    @Override
152    protected final Object getKey(final T resource) {
153      return Controller.this.getKey(resource);
154    }
155    
156  }
157    
158  private final class ControllerReflector extends Reflector<T> {
159
160    @SuppressWarnings("rawtypes")
161    private <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> ControllerReflector(final X operation,
162                                                                                                                                           final ScheduledExecutorService synchronizationExecutorService,
163                                                                                                                                           final Duration synchronizationInterval) {
164      super(operation, Controller.this.eventCache, synchronizationExecutorService, synchronizationInterval);
165    }
166    
167    @Override
168    protected final boolean shouldSynchronize() {
169      return Controller.this.shouldSynchronize();
170    }
171
172    @Override
173    protected final void onClose() {
174      Controller.this.onClose();
175    }
176  }
177  
178}
