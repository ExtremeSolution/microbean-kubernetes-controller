039/**
040 * An {@link EventCache} that temporarily stores {@link Event}s in
041 * {@link EventQueue}s, one per named Kubernetes resource, and
042 * {@linkplain #start(Consumer) provides a means for processing those
043 * queues}.
044 *
045 * <h2>Thread Safety</h2>
046 *
047 * <p>This class is safe for concurrent use by multiple {@link
048 * Thread}s.  Some operations, like the usage of the {@link
049 * #iterator()} method, require that callers synchronize on the {@link
050 * EventQueue} directly.  This class' internals synchronize on {@code
051 * this} when locking is needed.</p>
052 *
053 * @param <T> a type of Kubernetes resource
054 *
055 * @author <a href="https://about.me/lairdnelson"
056 * target="_parent">Laird Nelson</a>
057 *
058 * @see #add(Object, Event.Type, HasMetadata)
059 *
060 * @see #replace(Collection, Object)
061 *
062 * @see #synchronize()
063 *
064 * @see #start(Consumer)
065 *
066 * @see EventQueue
067 */
068public final class EventQueueCollection<T extends HasMetadata> implements EventCache<T>, Iterable<EventQueue<T>>, AutoCloseable {

304  /**
305   * <strong>Synchronizes on the {@code knownObjects} object</strong>
306   * {@linkplain #EventQueueCollection(Map, int, float) supplied at
307   * construction time}, if there is one, and, for every Kubernetes
308   * resource found within at the time of this call, {@linkplain
309   * #add(Object, Event.Type, HasMetadata) adds an <code>Event</code>}
310   * with an {@link Event.Type} of {@link Event.Type#SYNCHRONIZATION}
311   * for it.
312   */
313  @Override
314  public synchronized final void synchronize() {
315    final Map<?, ? extends T> knownObjects = this.getKnownObjects();
316    if (knownObjects != null) {
317      synchronized (knownObjects) {
318        if (!knownObjects.isEmpty()) {
319          final Collection<? extends T> values = knownObjects.values();
320          if (values != null && !values.isEmpty()) {
321            for (final T knownObject : values) {
322              if (knownObject != null) {
323                // We follow the Go code in that we use *our* key
324                // extraction logic, rather than relying on the known
325                // key in the knownObjects map.
326                final Object key = this.getKey(knownObject);
327                if (key != null) {
328                  final EventQueue<T> eventQueue = this.map.get(key);
329                  if (eventQueue == null || eventQueue.isEmpty()) {
330                    this.add(this, Event.Type.SYNCHRONIZATION, knownObject);
331                  }
332                }
333              }
334            }
335          }
336        }
337      }
338    }
339  }

341  /**
342   * At a high level, fully replaces the internal state of this {@link
343   * EventQueueCollection} to reflect only the Kubernetes resources
344   * contained in the supplied {@link Collection}, notionally firing
345   * {@link Event}s of type {@link Event.Type#SYNCHRONIZATION} and
346   * {@link Event.Type#DELETION} as appropriate.
347   *
348   * <p>{@link EventQueue}s managed by this {@link
349   * EventQueueCollection} that have not yet {@linkplain
350   * #start(Consumer) been processed} are not removed by this
351   * operation.</p>
352   *
353   * <p><strong>This method synchronizes on the supplied {@code
354   * incomingResources} {@link Collection}</strong> while iterating
355   * over it.</p>
356   *
357   * @param incomingResources the {@link Collection} of Kubernetes
358   * resources with which to replace this {@link
359   * EventQueueCollection}'s internal state; may be {@code null} or
360   * {@linkplain Collection#isEmpty() empty}, which will be taken as
361   * an indication that this {@link EventQueueCollection} should
362   * effectively be emptied
363   *
364   * @param resourceVersion the version of the Kubernetes list
365   * resource that contained the incoming resources; currently
366   * ignored; may be {@code null}
367   *
368   * @exception IllegalStateException if the {@link
369   * #createEvent(Object, Event.Type, HasMetadata)} method returns
370   * {@code null} for any reason
371   *
372   * @see Event.Type#SYNCHRONIZATION
373   *
374   * @see #createEvent(Object, Event.Type, HasMetadata)
375   */
376  @Override
377  public synchronized final void replace(final Collection<? extends T> incomingResources, final Object resourceVersion) {
378    
379    final int size;
380    final Set<Object> replacementKeys;
381
382    if (incomingResources == null) {
383      size = 0;
384      replacementKeys = Collections.emptySet();
385    } else {
386      synchronized (incomingResources) {
387        if (incomingResources.isEmpty()) {
388          size = 0;
389          replacementKeys = Collections.emptySet();
390        } else {
391          size = incomingResources.size();
392          assert size > 0;
393          replacementKeys = new HashSet<>();
394          for (final T resource : incomingResources) {
395            if (resource != null) {
396              replacementKeys.add(this.getKey(resource));
397              this.add(this, Event.Type.SYNCHRONIZATION, resource, false);
398            }
399          }
400        }
401      }
402    }
403
404    int queuedDeletions = 0;
405    
406    final Map<?, ? extends T> knownObjects = this.getKnownObjects();
407    if (knownObjects == null) {
408
409      for (final EventQueue<T> eventQueue : this.map.values()) {
410        assert eventQueue != null;
411        final Object key;
412        final Event<? extends T> newestEvent;
413        synchronized (eventQueue) {
414          if (eventQueue.isEmpty()) {
415            newestEvent = null;
416            key = null;
417            assert false : "eventQueue.isEmpty(): " + eventQueue;
418          } else {
419            key = eventQueue.getKey();
420            if (key == null) {
421              throw new IllegalStateException();
422            }
423            if (replacementKeys.contains(key)) {
424              newestEvent = null;
425            } else {
426              // We have an EventQueue indexed under a key that
427              // identifies a resource that no longer exists in
428              // Kubernetes.  Inform any consumers via a deletion
429              // event that this object was removed at some point from
430              // Kubernetes.  The state of the object in question is
431              // indeterminate.
432              newestEvent = eventQueue.getLast();
433              assert newestEvent != null;
434            }
435          }
436        }
437        if (newestEvent != null) {
438          assert key != null;
439          // We grab the last event in the queue in question and get
440          // his resource; this will serve as the state of the
441          // Kubernetes resource in question the last time we knew
442          // about it.  This state is not necessarily, but could be,
443          // the true actual last state of the resource in question.
444          // The point is, the true state of the object when it was
445          // deleted is unknown.  We build a new event to reflect all
446          // this.
447          //
448          // Astute readers will realize that this could result in two
449          // DELETION events enqueued, back to back, with identical
450          // payloads.  See the deduplicate() method in EventQueue.
451          final T resourceToBeDeleted = newestEvent.getResource();
452          assert resourceToBeDeleted != null;
453          final Event<T> event = this.createEvent(this, Event.Type.DELETION, resourceToBeDeleted);
454          if (event == null) {
455            throw new IllegalStateException("createEvent() == null");
456          }
457          event.setKey(key);
458          this.add(event, false);
459        }
460      }
461      
462    } else {
463
464      synchronized (knownObjects) {
465        if (!knownObjects.isEmpty()) {
466          final Collection<? extends Entry<?, ? extends T>> entrySet = knownObjects.entrySet();
467          if (entrySet != null && !entrySet.isEmpty()) {
468            for (final Entry<?, ? extends T> entry : entrySet) {
469              if (entry != null) {
470                final Object knownKey = entry.getKey();
471                if (!replacementKeys.contains(knownKey)) {
472                  final Event<T> event = this.createEvent(this, Event.Type.DELETION, entry.getValue());
473                  if (event == null) {
474                    throw new IllegalStateException("createEvent() == null");
475                  }
476                  event.setKey(knownKey);
477                  this.add(event, false);
478                  queuedDeletions++;
479                }
480              }
481            }
482          }
483        }
484      }
485      
486    }
487      
488    if (!this.populated) {
489      this.populated = true;
490      this.initialPopulationCount = size + queuedDeletions;
491    }
492    
493  }

495  /**
496   * Returns an {@link Object} which will be used as the key that will
497   * uniquely identify the supplied {@code resource} to this {@link
498   * EventQueueCollection}.
499   *
500   * <p>This method may return {@code null}, but only if {@code
501   * resource} is {@code null} or is constructed in such a way that
502   * its {@link HasMetadata#getMetadata()} method returns {@code
503   * null}.</p>
504   *
505   * <p>Overrides of this method may return {@code null}, but only if
506   * {@code resource} is {@code null}.
507   *
508   * @param resource a {@link HasMetadata} for which a key should be
509   * returned; may be {@code null} in which case {@code null} may be
510   * returned
511   *
512   * @return a non-{@code null} key for the supplied {@code resource};
513   * or {@code null} if {@code resource} is {@code null}
514   */
515  protected Object getKey(final T resource) {
516    final Object returnValue;
517    if (resource == null) {
518      returnValue = null;
519    } else {
520      final ObjectMeta metadata = resource.getMetadata();
521      if (metadata == null) {
522        returnValue = null;
523      } else {
524        String name = metadata.getName();
525        if (name == null) {
526          throw new IllegalStateException("metadata.getName() == null");
527        } else if (name.isEmpty()) {
528          throw new IllegalStateException("metadata.getName().isEmpty()");
529        }
530        final String namespace = metadata.getNamespace();
531        if (namespace == null || namespace.isEmpty()) {
532          returnValue = name;
533        } else {
534          returnValue = new StringBuilder(namespace).append("/").append(name).toString();
535        }
536      }
537    }
538    return returnValue;
539  }

563  /**
564   * Starts a new {@linkplain Thread#isDaemon() daemon} {@link Thread}
565   * that, until {@link #close()} is called, removes {@link
566   * EventQueue}s from this {@link EventQueueCollection} and supplies
567   * them to the supplied {@link Consumer}.
568   *
569   * <p>Invoking this method does not block the calling {@link
570   * Thread}.</p>
571   *
572   * @param siphon the {@link Consumer} that will process each {@link
573   * EventQueue} as it becomes ready; must not be {@code null}
574   *
575   * @exception NullPointerException if {@code siphon} is {@code null}
576   */
577  @NonBlocking
578  public final void start(final Consumer<? super EventQueue<? extends T>> siphon) {
579    Objects.requireNonNull(siphon);
580    synchronized (this) {
581      if (this.consumerThread == null) {
582        this.consumerThread = new Thread(() -> {
583            try {
584              attach(siphon);
585            } catch (final InterruptedException interruptedException) {
586              Thread.currentThread().interrupt();
587            }
588          });
589        this.consumerThread.setDaemon(true);
590        this.consumerThread.start();
591      }
592    }
593  }

595  /**
596   * Semantically closes this {@link EventQueueCollection} by
597   * detaching any {@link Consumer} previously attached via the {@link
598   * #start(Consumer)} method.  {@linkplain #add(Object, Event.Type,
599   * HasMetadata) Additions}, {@linkplain #replace(Collection, Object)
600   * replacements} and {@linkplain #synchronize() synchronizations}
601   * are still possible, but there won't be anything consuming any
602   * events generated by or supplied to these operations.
603   *
604   * <p>A closed {@link EventQueueCollection} may be {@linkplain
605   * #start(Consumer) started} again.</p>
606   *
607   * @see #start(Consumer)
608   */
609  @Override
610  public final void close() {
611    final Thread consumerThread;
612    synchronized (this) {
613      consumerThread = this.consumerThread;
614      if (consumerThread != null) {
615        this.closing = true;
616        this.consumerThread.interrupt();
617      }
618    }
619    if (consumerThread != null) {
620      try {
621        consumerThread.join();
622      } catch (final InterruptedException interruptedException) {
623        Thread.currentThread().interrupt();
624      } finally {
625        synchronized (this) {
626          this.consumerThread = null;
627          this.closing = false;
628        }
629      }
630    }
631  }

633  /**
634   * <strong>Blocks the current thread</strong> and removes and
635   * funnels current and future {@link EventQueue}s belonging to this
636   * {@link EventQueueCollection} into the supplied {@link Consumer},
637   * blocking if necessary until there are new {@link EventQueue}s in
638   * this {@link EventQueueCollection} for the supplied {@link
639   * Consumer} to consume.
640   *
641   * <p>The supplied {@link Consumer} may call any method on the
642   * {@link EventQueue} supplied to it without synchronization as the
643   * {@link EventQueue}'s monitor is obtained by the {@link Thread}
644   * invoking the {@link Consumer}.</p>
645   *
646   * @param siphon a {@link Consumer} that can process an {@link
647   * EventQueue} or its superclasses; must not be {@code null}
648   *
649   * @exception NullPointerException if {@code consumer} is {@code null}
650   *
651   * @exception InterruptedException if the {@link Thread} currently
652   * occupied with this method invocation was {@linkplain
653   * Thread#interrupt() interrupted}
654   */
655  @Blocking
656  private final void attach(final Consumer<? super EventQueue<? extends T>> siphon) throws InterruptedException {
657    Objects.requireNonNull(siphon);
658    while (!this.closing) {
659      synchronized (this) {
660        @Blocking
661        final EventQueue<T> eventQueue = this.take();
662        if (eventQueue != null) {
663          synchronized (eventQueue) {
664            try {
665              siphon.accept(eventQueue);
666            } catch (final RuntimeException runtimeException) {
667              // TODO: catch a subclass that indicates a requeue request;
668              throw runtimeException;
669            }
670          }
671        }
672      }
673    }
674  }
675
676  @Blocking
677  private synchronized final EventQueue<T> take() throws InterruptedException {
678    while (!this.closing && this.isEmpty()) {
679      try {
680        this.wait();
681      } catch (final InterruptedException interruptedException) {
682        if (this.closing) {
683          throw interruptedException;
684        }
685        Thread.currentThread().interrupt();
686      }
687    }
688    final EventQueue<T> returnValue;
689    if (this.closing) {
690      returnValue = null;
691    } else {
692      assert !this.isEmpty();
693      final Iterator<EventQueue<T>> iterator = this.map.values().iterator();
694      assert iterator != null;
695      assert iterator.hasNext();
696      returnValue = iterator.next();
697      assert returnValue != null;
698      iterator.remove();
699      if (this.initialPopulationCount > 0) {
700        this.initialPopulationCount--;
701      }
702      Thread.interrupted(); // clear any interrupted status now that we know we're going to be successful
703    }
704    return returnValue;
705  }

715  public final Event<T> add(final Object source, final Event.Type eventType, final T resource) {
716    return this.add(source, eventType, resource, true);
717  }
718
719  private final Event<T> add(final Object source, final Event.Type eventType, final T resource, final boolean populate) {
720    final Event<T> event = this.createEvent(source, eventType, resource);
721    if (event == null) {
722      throw new IllegalStateException("createEvent() == null");
723    }
724    return this.add(event, populate);
725  }
726
727  private final Event<T> add(final Event<T> event, final boolean populate) {
728    Objects.requireNonNull(event);
729    final Object key = event.getKey();
730    if (key == null) {
731      throw new IllegalArgumentException("event.getKey() == null");
732    }
733    if (populate) {
734      this.populated = true;
735    }
736    
737    Event<T> returnValue = null;
738    EventQueue<T> eventQueue;
739    synchronized (this) {
740      eventQueue = this.map.get(key);
741      final boolean eventQueueExisted = eventQueue != null;
742      if (eventQueue == null) {
743        eventQueue = this.createEventQueue(key);
744        if (eventQueue == null) {
745          throw new IllegalStateException("createEventQueue(key) == null: " + key);
746        }
747      }
748      synchronized (eventQueue) {
749        if (eventQueue.addEvent(event)) {
750          returnValue = event;
751        }
752        if (eventQueue.isEmpty()) {
753          // Compression might have emptied the queue, so an add could
754          // result in an empty queue.
755          if (eventQueueExisted) {
756            returnValue = null;
757            this.map.remove(key);
758          }
759        } else if (!eventQueueExisted) {
760          this.map.put(key, eventQueue);
761        }
762      }
763      this.notifyAll();
764    }
765    return returnValue;
766  }
