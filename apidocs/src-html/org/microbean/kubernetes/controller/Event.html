017package org.microbean.kubernetes.controller;
018
019import java.io.Serializable; // for javadoc only
020
021import java.util.EventObject;
022import java.util.Objects;
023
024import io.fabric8.kubernetes.api.model.HasMetadata;
025import io.fabric8.kubernetes.api.model.ObjectMeta;

027/**
028 * An {@link EventObject} that represents another event that has
029 * occurred to a Kubernetes resource, usually as found in an {@link
030 * EventCache} implementation.
031 *
032 * @param <T> a type of Kubernetes resource
033 *
034 * @author <a href="https://about.me/lairdnelson"
035 * target="_parent">Laird Nelson</a>
036 *
037 * @see EventCache
038 */
039public class Event<T extends HasMetadata> extends EventObject {
040
041
042  /*
043   * Static fields.
044   */
045
046
047  /**
048   * The version of this class for {@linkplain Serializable
049   * serialization purposes}.
050   *
051   * @see Serializable
052   */
053  private static final long serialVersionUID = 1L;
054
055
056  /*
057   * Instance fields.
058   */
059  
060
061  /**
062   * The key that identifies this {@link Event}'s {@linkplain
063   * #getResource() resource} <strong>only when its final state is
064   * unknown</strong>.
065   *
066   * <p>This field can be&mdash;and often is&mdash;{@code null}.</p>
067   *
068   * @see #getKey()
069   *
070   * @see #setKey(Object)
071   */
072  private volatile Object key;
073
074  /**
075   * The {@link Type} describing the type of this {@link Event}.
076   *
077   * <p>This field is never {@code null}.</p>
078   *
079   * @see #getType()
080   */
081  private final Type type;
082
083  /**
084   * A Kubernetes resource representing its state at the time of this
085   * event.
086   *
087   * <p>This field is never {@code null}.</p>
088   *
089   * @see #getResource()
090   */
091  private final T resource;
092
093
094  /*
095   * Constructors.
096   */
097
098
099  /**
100   * Creates a new {@link Event}.
101   *
102   * @param source the creator; must not be {@code null}
103   *
104   * @param type the {@link Type} of this {@link Event}; must not be
105   * {@code null}
106   *
107   * @param resource a {@link HasMetadata} representing a Kubernetes
108   * resource; must not be {@code null}
109   *
110   * @see Type
111   *
112   * @see EventObject#getSource()
113   */
114  protected Event(final Object source, final Type type, final T resource) {
115    super(source);
116    this.type = Objects.requireNonNull(type);
117    this.resource = Objects.requireNonNull(resource);
118  }
119
120
121  /*
122   * Instance methods.
123   */
124
125
126  /**
127   * Returns a {@link Type} representing the type of this {@link
128   * Event}.
129   *
130   * <p>This method never returns {@code null}.</p>
131   *
132   * @return a non-{@code null} {@link Type}
133   *
134   * @see Type
135   */
136  public final Type getType() {
137    return this.type;
138  }
139
140  /**
141   * Returns a {@link HasMetadata} representing the Kubernetes
142   * resource this {@link Event} concerns.
143   *
144   * <p>This method never returns {@code null}.</p>
145   *
146   * @return a non-{@code null} Kubernetes resource
147   */
148  public final T getResource() {
149    return this.resource;
150  }
151
152  /**
153   * Returns {@code true} if this {@link Event}'s {@linkplain
154   * #getResource() resource} is an accurate representation of its
155   * last known state.
156   *
157   * @return {@code true} if this {@link Event}'s {@linkplain
158   * #getResource() resource} is an accurate representation of its
159   * last known state; {@code false} otherwise
160   */
161  public final boolean isFinalStateKnown() {
162    return this.key == null;
163  }
164
165  /**
166   * Sets the key identifying the Kubernetes resource this {@link
167   * Event} describes.
168   *
169   * @param key the new key; may be {@code null}
170   *
171   * @see #getKey()
172   */
173  final void setKey(final Object key) {
174    this.key = key;
175  }
176
177  /**
178   * Returns a key that can be used to unambiguously identify this
179   * {@link Event}'s {@linkplain #getResource() resource}.
180   *
181   * <p>This method may return {@code null} in exceptional cases, but
182   * normally does not.</p>
183   *
184   * <p>Overrides of this method must not return {@code null} except
185   * in exceptional cases.</p>
186   *
187   * @return a key for this {@link Event}, or {@code null}
188   */
189  public Object getKey() {
190    Object returnValue = this.key;
191    if (returnValue == null) {
192      final T resource = this.getResource();
193      assert resource != null;
194      final ObjectMeta metadata = resource.getMetadata();
195      if (metadata != null) {
196        final String namespace = metadata.getNamespace();
197        if (namespace == null || namespace.isEmpty()) {
198          returnValue = metadata.getName();
199        } else {
200          returnValue = new StringBuilder(namespace).append("/").append(metadata.getName()).toString();
201        }
202      }
203    }
204    return returnValue;
205  }
206
207  /**
208   * Returns a hashcode for this {@link Event}.
209   *
210   * @return a hashcode for this {@link Event}
211   */
212  @Override
213  public int hashCode() {
214    int hashCode = 37;
215    
216    final Object source = this.getSource();
217    int c = source == null ? 0 : source.hashCode();
218    hashCode = hashCode * 17 + c;
219    
220    final Object key = this.getKey();
221    c = key == null ? 0 : key.hashCode();
222    hashCode = hashCode * 17 + c;
223    
224    final Object type = this.getType();
225    c = type == null ? 0 : type.hashCode();
226    hashCode = hashCode * 17 + c;
227    
228    final Object resource = this.getResource();
229    c = resource == null ? 0 : resource.hashCode();
230    hashCode = hashCode * 17 + c;
231    
232    return hashCode;
233  }
234
235  /**
236   * Returns {@code true} if the supplied {@link Object} is also an
237   * {@link Event} and is equal in every respect to this one.
238   *
239   * @param other the {@link Object} to test; may be {@code null} in
240   * which case {@code false} will be returned
241   *
242   * @return {@code true} if the supplied {@link Object} is also an
243   * {@link Event} and is equal in every respect to this one; {@code
244   * false} otherwise
245   */
246  @Override
247  public boolean equals(final Object other) {
248    if (other == this) {
249      return true;
250    } else if (other instanceof Event) {
251      
252      final Event<?> her = (Event<?>)other;
253      
254      final Object source = this.getSource();
255      if (source == null) {
256        if (her.getSource() != null) {
257          return false;
258        }
259      } else if (!source.equals(her.getSource())) {
260        return false;
261      }
262      
263      final Object key = this.getKey();
264      if (key == null) {
265        if (her.getKey() != null) {
266          return false;
267        }
268      } else if (!key.equals(her.getKey())) {
269        return false;
270      }
271      
272      final Object type = this.getType();
273      if (type == null) {
274        if (her.getType() != null) {
275          return false;
276        }
277      } else if (!type.equals(her.getType())) {
278        return false;
279      }
280      
281      final Object resource = this.getResource();
282      if (resource == null) {
283        if (her.getResource() != null) {
284          return false;
285        }
286      } else if (!resource.equals(her.getResource())) {
287        return false;
288      }
289
290      return true;
291    } else {
292      return false;
293    }
294  }
295
296  /**
297   * Returns a {@link String} representation of this {@link Event}.
298   *
299   * <p>This method never returns {@code null}.</p>
300   *
301   * <p>Overrides of this method must not return {@code null}.</p>
302   *
303   * @return a non-{@code null} {@link String} representation of this
304   * {@link Event}
305   */
306  @Override
307  public String toString() {
308    return new StringBuilder().append(this.getType()).append(": ").append(this.getResource()).toString();
309  }

344    SYNCHRONIZATION
