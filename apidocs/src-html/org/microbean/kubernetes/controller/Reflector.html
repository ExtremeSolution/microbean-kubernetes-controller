051/**
052 * A pump of sorts that continuously "pulls" logical events out of
053 * Kubernetes and {@linkplain EventCache#add(Object, Event.Type,
054 * HasMetadata) adds them} to an {@link EventCache} so as to logically
055 * "reflect" the contents of Kubernetes into the cache.
056 *
057 * <h2>Thread Safety</h2>
058 *
059 * <p>Instances of this class are safe for concurrent use by multiple
060 * {@link Thread}s.</p>
061 *
062 * @param <T> a type of Kubernetes resource
063 *
064 * @author <a href="https://about.me/lairdnelson"
065 * target="_parent">Laird Nelson</a>
066 *
067 * @see EventCache
068 */
069public class Reflector<T extends HasMetadata> implements Closeable {

171  /**
172   * Creates a new {@link Reflector}.
173   *
174   * @param <X> a type that is both an appropriate kind of {@link
175   * Listable} and {@link VersionWatchable}, such as the kind of
176   * operation returned by {@link
177   * DefaultKubernetesClient#configMaps()} and the like
178   *
179   * @param operation a {@link Listable} and a {@link
180   * VersionWatchable} that can report information from a Kubernetes
181   * cluster; must not be {@code null}
182   *
183   * @param eventCache an {@link EventCache} <strong>that will be
184   * synchronized on</strong> and into which {@link Event}s will be
185   * logically "reflected"; must not be {@code null}
186   *
187   * @param synchronizationExecutorService a {@link
188   * ScheduledExecutorService} to be used to tell the supplied {@link
189   * EventCache} to {@linkplain EventCache#synchronize() synchronize}
190   * on a schedule; may be {@code null} in which case no
191   * synchronization will occur
192   *
193   * @param synchronizationInterval a {@link Duration} representing
194   * the time in between one {@linkplain EventCache#synchronize()
195   * synchronization operation} and another; may be {@code null} in
196   * which case no synchronization will occur
197   *
198   * @see #start()
199   */
200  @SuppressWarnings("rawtypes") // kubernetes-client's implementations of KubernetesResourceList use raw types
201  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Reflector(final X operation,
202                                                                                                                              final EventCache<T> eventCache,
203                                                                                                                              final ScheduledExecutorService synchronizationExecutorService,
204                                                                                                                              final Duration synchronizationInterval) {
205    super();
206    Objects.requireNonNull(operation);
207    this.eventCache = Objects.requireNonNull(eventCache);
208    // TODO: research: maybe: operation.withField("metadata.resourceVersion", "0")?    
209    this.operation = operation.withResourceVersion("0");
210    this.synchronizationInterval = synchronizationInterval;
211    if (synchronizationExecutorService == null) {
212      if (synchronizationInterval == null) {
213        this.synchronizationExecutorService = null;
214        this.shutdownSynchronizationExecutorServiceOnClose = false;
215      } else {
216        this.synchronizationExecutorService = Executors.newScheduledThreadPool(1);
217        this.shutdownSynchronizationExecutorServiceOnClose = true;
218      }
219    } else {
220      this.synchronizationExecutorService = synchronizationExecutorService;
221      this.shutdownSynchronizationExecutorServiceOnClose = false;
222    }
223  }

231  /**
232   * Notionally closes this {@link Reflector} by terminating any
233   * {@link Thread}s that it has started and invoking the {@link
234   * #onClose()} method while holding this {@link Reflector}'s
235   * monitor.
236   *
237   * @exception IOException if an error occurs
238   *
239   * @see #onClose()
240   */
241  @Override
242  public synchronized final void close() throws IOException {
243    try {
244      final ScheduledFuture<?> synchronizationTask = this.synchronizationTask;
245      if (synchronizationTask != null) {
246        synchronizationTask.cancel(false);
247      }
248      this.closeSynchronizationExecutorService();
249      if (this.watch != null) {
250        this.watch.close();
251      }
252    } finally {
253      this.onClose();
254    }
255  }
256
257  private synchronized final void closeSynchronizationExecutorService() {
258    if (this.synchronizationExecutorService != null && this.shutdownSynchronizationExecutorServiceOnClose) {
259      this.synchronizationExecutorService.shutdown();
260      try {
261        if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
262          this.synchronizationExecutorService.shutdownNow();
263          if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
264            this.synchronizationExecutorService.shutdownNow();
265          }
266        }
267      } catch (final InterruptedException interruptedException) {
268        this.synchronizationExecutorService.shutdownNow();
269        Thread.currentThread().interrupt();
270      }
271    }
272  }
273
274  private synchronized final void setUpSynchronization() {
275    if (this.synchronizationExecutorService != null) {
276      
277      final Duration synchronizationDuration = this.getSynchronizationInterval();
278      final long seconds;
279      if (synchronizationDuration == null) {
280        seconds = 0L;
281      } else {
282        seconds = synchronizationDuration.get(ChronoUnit.SECONDS);
283      }
284        
285      if (seconds > 0L) {
286        final ScheduledFuture<?> job = this.synchronizationExecutorService.scheduleWithFixedDelay(() -> {
287            try {
288              if (shouldSynchronize()) {
289                synchronized (eventCache) {
290                  eventCache.synchronize();
291                }
292              }
293            } catch (final RuntimeException runtimeException) {
294              // TODO: log or...?
295              throw runtimeException;
296            }
297          }, 0L, seconds, TimeUnit.SECONDS);
298        assert job != null;
299        this.synchronizationTask = job;
300      }
301      
302    }
303  }

305  /**
306   * Whether, at any given moment, this {@link Reflector} should cause
307   * its {@link EventCache} to {@linkplain EventCache#synchronize()
308   * synchronize}.
309   *
310   * <h2>Design Notes</h2>
311   *
312   * <p>This code follows the Go code in the Kubernetes {@code
313   * client-go/tools/cache} package.  One thing that becomes clear
314   * when looking at all of this through an object-oriented lens is
315   * that it is the {@link EventCache} (the {@code delta_fifo}, in the
316   * Go code) that is ultimately in charge of synchronizing.  It is
317   * not clear why this is a function of a reflector.  In an
318   * object-oriented world, perhaps the {@link EventCache} itself
319   * should be in charge of resynchronization schedules.</p>
320   *
321   * @return {@code true} if this {@link Reflector} should cause its
322   * {@link EventCache} to {@linkplain EventCache#synchronize()
323   * synchronize}; {@code false} otherwise
324   */
325  protected boolean shouldSynchronize() {
326    return this.synchronizationExecutorService != null;
327  }

334    return this.lastSynchronizationResourceVersion;

341  /**
342   * Using the {@code operation} supplied at construction time,
343   * {@linkplain Listable#list() lists} appropriate Kubernetes
344   * resources, and then, on a separate {@link Thread}, {@linkplain
345   * VersionWatchable sets up a watch} on them, calling {@link
346   * EventCache#replace(Collection, Object)} and {@link
347   * EventCache#add(Object, Event.Type, HasMetadata)} methods as
348   * appropriate.
349   *
350   * <p>The calling {@link Thread} is not blocked by invocations of
351   * this method.</p>
352   *
353   * @see #close()
354   */
355  @NonBlocking
356  public synchronized final void start() {
357    if (this.watch == null) {
358
359      // Run a list operation, and get the resourceVersion of that list.
360      @SuppressWarnings("unchecked")
361      final KubernetesResourceList<? extends T> list = ((Listable<? extends KubernetesResourceList<? extends T>>)this.operation).list();
362      assert list != null;
363      final ListMeta metadata = list.getMetadata();
364      assert metadata != null;      
365      final String resourceVersion = metadata.getResourceVersion();
366      assert resourceVersion != null;
367
368      // Using the results of that list operation, do a full replace
369      // on the EventCache with them.
370      final Collection<? extends T> replacementItems;
371      final Collection<? extends T> items = list.getItems();
372      if (items == null || items.isEmpty()) {
373        replacementItems = Collections.emptySet();
374      } else {
375        replacementItems = Collections.unmodifiableCollection(new ArrayList<>(items));
376      }
377      synchronized (eventCache) {
378        this.eventCache.replace(replacementItems, resourceVersion);
379      }
380
381      // Record the resource version we captured during our list
382      // operation.
383      this.setLastSynchronizationResourceVersion(resourceVersion);
384
385      // Now that we've vetted that our list operation works (i.e. no
386      // syntax errors, no connectivity problems) we can schedule
387      // resynchronizations if necessary.
388      this.setUpSynchronization();
389
390      // Now that we've taken care of our list() operation, set up our
391      // watch() operation.
392      try {
393        @SuppressWarnings("unchecked")
394        final Closeable temp = ((VersionWatchable<? extends Closeable, Watcher<T>>)operation).withResourceVersion(resourceVersion).watch(new WatchHandler());
395        assert temp != null;
396        this.watch = temp;
397      } finally {
398        this.closeSynchronizationExecutorService();
399      }
400    }
401  }

447    /**
448     * Calls the {@link EventCache#add(Object, Event.Type,
449     * HasMetadata)} method on the enclosing {@link Reflector}'s
450     * associated {@link EventCache} with information harvested from
451     * the supplied {@code resource}, and using an {@link Event.Type}
452     * selected appropriately given the supplied {@link
453     * Watcher.Action}.
454     *
455     * @param action the kind of Kubernetes event that happened; must
456     * not be {@code null}
457     *
458     * @param resource the {@link HasMetadata} object that was
459     * affected; must not be {@code null}
460     *
461     * @exception NullPointerException if {@code action} or {@code
462     * resource} was {@code null}
463     *
464     * @exception IllegalStateException if another error occurred
465     */
466    @Override
467    public final void eventReceived(final Watcher.Action action, final T resource) {
468      Objects.requireNonNull(action);
469      Objects.requireNonNull(resource);
470
471      final ObjectMeta metadata = resource.getMetadata();
472      assert metadata != null;
473
474      final Event.Type eventType;
475
476      switch (action) {
477      case ADDED:
478        eventType = Event.Type.ADDITION;
479        break;
480      case MODIFIED:
481        eventType = Event.Type.MODIFICATION;
482        break;
483      case DELETED:
484        eventType = Event.Type.DELETION;
485        break;
486      case ERROR:        
487        // TODO: Uh...the Go code has:
488        //
489        //   if event.Type == watch.Error {
490        //     return apierrs.FromObject(event.Object)
491        //   }
492        //
493        // Now, apierrs.FromObject is here:
494        // https://github.com/kubernetes/apimachinery/blob/kubernetes-1.9.2/pkg/api/errors/errors.go#L80-L88
495        // This is looking for a Status object.  But
496        // WatchConnectionHandler will never forward on such a thing:
497        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L246-L258
498        //
499        // So it follows that if by some chance we get here, resource
500        // will definitely be a HasMetadata.  We go back to the Go
501        // code again, and remember that if the type is Error, the
502        // equivalent of this watch handler simply returns and goes home.
503        //
504        // Now, if we were to throw a RuntimeException here, which is
505        // the idiomatic equivalent of returning and going home, this
506        // would cause a watch reconnect:
507        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L159-L205
508        // ...up to the reconnect limit.
509        //
510        // ...which is fine, but I'm not sure that in an error case a
511        // WatchEvent will ever HAVE a HasMetadata as its payload.
512        // Which means MAYBE we'll never get here.  But if we do, all
513        // we can do is throw a RuntimeException...which ends up
514        // reducing to the same case as the default case below, so we
515        // fall through.
516      default:
517        eventType = null;
518        throw new IllegalStateException();
519      }
520
521      // Add an Event of the proper kind to our EventCache.
522      if (eventType != null) {
523        synchronized (eventCache) {
524          eventCache.add(Reflector.this, eventType, resource);
525        }
526      }
527
528      // Record the most recent resource version we're tracking to be
529      // that of this last successful watch() operation.  We set it
530      // earlier during a list() operation.
531      setLastSynchronizationResourceVersion(metadata.getResourceVersion());
532    }
