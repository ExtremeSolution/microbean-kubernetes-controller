057/**
058 * A pump of sorts that continuously "pulls" logical events out of
059 * Kubernetes and {@linkplain EventCache#add(Object, AbstractEvent.Type,
060 * HasMetadata) adds them} to an {@link EventCache} so as to logically
061 * "reflect" the contents of Kubernetes into the cache.
062 *
063 * <h2>Thread Safety</h2>
064 *
065 * <p>Instances of this class are safe for concurrent use by multiple
066 * {@link Thread}s.</p>
067 *
068 * @param <T> a type of Kubernetes resource
069 *
070 * @author <a href="https://about.me/lairdnelson"
071 * target="_parent">Laird Nelson</a>
072 *
073 * @see EventCache
074 */
075@ThreadSafe
076public class Reflector<T extends HasMetadata> implements Closeable {

209  /**
210   * Creates a new {@link Reflector}.
211   *
212   * @param <X> a type that is both an appropriate kind of {@link
213   * Listable} and {@link VersionWatchable}, such as the kind of
214   * operation returned by {@link
215   * DefaultKubernetesClient#configMaps()} and the like
216   *
217   * @param operation a {@link Listable} and a {@link
218   * VersionWatchable} that can report information from a Kubernetes
219   * cluster; must not be {@code null}
220   *
221   * @param eventCache an {@link EventCache} <strong>that will be
222   * synchronized on</strong> and into which {@link Event}s will be
223   * logically "reflected"; must not be {@code null}
224   *
225   * @param synchronizationExecutorService a {@link
226   * ScheduledExecutorService} to be used to tell the supplied {@link
227   * EventCache} to {@linkplain EventCache#synchronize() synchronize}
228   * on a schedule; may be {@code null} in which case no
229   * synchronization will occur
230   *
231   * @param synchronizationInterval a {@link Duration} representing
232   * the time in between one {@linkplain EventCache#synchronize()
233   * synchronization operation} and another; may be {@code null} in
234   * which case no synchronization will occur
235   *
236   * @exception NullPointerException if {@code operation} or {@code
237   * eventCache} is {@code null}
238   *
239   * @exception IllegalStateException if the {@link #createLogger()}
240   * method returns {@code null}
241   *
242   * @see #start()
243   */
244  @SuppressWarnings("rawtypes") // kubernetes-client's implementations of KubernetesResourceList use raw types
245  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Reflector(final X operation,
246                                                                                                                              final EventCache<T> eventCache,
247                                                                                                                              final ScheduledExecutorService synchronizationExecutorService,
248                                                                                                                              final Duration synchronizationInterval) {
249    super();
250    this.logger = this.createLogger();
251    if (this.logger == null) {
252      throw new IllegalStateException("createLogger() == null");
253    }
254    final String cn = this.getClass().getName();
255    final String mn = "<init>";
256    if (this.logger.isLoggable(Level.FINER)) {
257      this.logger.entering(cn, mn, new Object[] { operation, eventCache, synchronizationExecutorService, synchronizationInterval });
258    }
259    Objects.requireNonNull(operation);
260    this.eventCache = Objects.requireNonNull(eventCache);
261    // TODO: research: maybe: operation.withField("metadata.resourceVersion", "0")?    
262    this.operation = operation.withResourceVersion("0");
263    if (synchronizationInterval == null) {
264      this.synchronizationIntervalInSeconds = 0L;
265    } else {
266      this.synchronizationIntervalInSeconds = synchronizationInterval.get(ChronoUnit.SECONDS);
267    }
268    if (this.synchronizationIntervalInSeconds <= 0L) {
269      this.synchronizationExecutorService = null;
270      this.shutdownSynchronizationExecutorServiceOnClose = false;
271    } else if (synchronizationExecutorService == null) {
272      this.synchronizationExecutorService = Executors.newScheduledThreadPool(1);
273      this.shutdownSynchronizationExecutorServiceOnClose = true;
274    } else {
275      this.synchronizationExecutorService = synchronizationExecutorService;
276      this.shutdownSynchronizationExecutorServiceOnClose = false;
277    }
278    if (this.logger.isLoggable(Level.FINER)) {
279      this.logger.exiting(cn, mn);
280    }
281  }

303  /**
304   * Notionally closes this {@link Reflector} by terminating any
305   * {@link Thread}s that it has started and invoking the {@link
306   * #onClose()} method while holding this {@link Reflector}'s
307   * monitor.
308   *
309   * @exception IOException if an error occurs
310   *
311   * @see #onClose()
312   */
313  @Override
314  public synchronized final void close() throws IOException {
315    final String cn = this.getClass().getName();
316    final String mn = "close";
317    if (this.logger.isLoggable(Level.FINER)) {
318      this.logger.entering(cn, mn);
319    }
320    try {
321      this.closeSynchronizationExecutorService();
322      if (this.watch != null) {
323        this.watch.close();
324      }
325    } finally {
326      this.onClose();
327    }
328    if (this.logger.isLoggable(Level.FINER)) {
329      this.logger.exiting(cn, mn);
330    }
331  }

333  private synchronized final void closeSynchronizationExecutorService() {
334    final String cn = this.getClass().getName();
335    final String mn = "closeSynchronizationExecutorService";
336    if (this.logger.isLoggable(Level.FINER)) {
337      this.logger.entering(cn, mn);
338    }
339    if (this.synchronizationExecutorService != null && this.shutdownSynchronizationExecutorServiceOnClose) {
340
341      // Stop accepting new tasks.  Not that any will be showing up
342      // anyway, but it's the right thing to do.
343      this.synchronizationExecutorService.shutdown();
344
345      // Nicely cancel our task.
346      final ScheduledFuture<?> synchronizationTask = this.synchronizationTask;
347      if (synchronizationTask != null) {
348        synchronizationTask.cancel(true /* interrupt the task */);
349      }
350
351      try {
352        if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
353          this.synchronizationExecutorService.shutdownNow();
354          if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
355            // TODO: log
356          }
357        }
358      } catch (final InterruptedException interruptedException) {
359        this.synchronizationExecutorService.shutdownNow();
360        Thread.currentThread().interrupt();
361      }
362    }
363    if (this.logger.isLoggable(Level.FINER)) {
364      this.logger.exiting(cn, mn);
365    }
366  }

368  private synchronized final void setUpSynchronization() {
369    final String cn = this.getClass().getName();
370    final String mn = "setUpSynchronization";
371    if (this.logger.isLoggable(Level.FINER)) {
372      this.logger.entering(cn, mn);
373    }
374
375    if (this.synchronizationExecutorService != null && this.synchronizationTask == null) {
376      assert this.synchronizationIntervalInSeconds > 0L;
377      
378      if (this.logger.isLoggable(Level.INFO)) {
379        this.logger.logp(Level.INFO, cn, mn, "Scheduling downstream synchronization every {0} seconds", Long.valueOf(this.synchronizationIntervalInSeconds));
380      }
381      final ScheduledFuture<?> job = this.synchronizationExecutorService.scheduleWithFixedDelay(() -> {
382          if (shouldSynchronize()) {
383            if (logger.isLoggable(Level.FINE)) {
384              logger.logp(Level.FINE, cn, mn, "Synchronizing event cache with its downstream consumers");
385            }
386            synchronized (eventCache) {
387              eventCache.synchronize();
388            }
389          }
390        }, 0L, this.synchronizationIntervalInSeconds, TimeUnit.SECONDS);
391      assert job != null;
392      this.synchronizationTask = job;
393    }
394
395    if (this.logger.isLoggable(Level.FINER)) {
396      this.logger.exiting(cn, mn);
397    }
398  }

400  /**
401   * Returns whether, at any given moment, this {@link Reflector}
402   * should cause its {@link EventCache} to {@linkplain
403   * EventCache#synchronize() synchronize}.
404   *
405   * <h2>Design Notes</h2>
406   *
407   * <p>This code follows the Go code in the Kubernetes {@code
408   * client-go/tools/cache} package.  One thing that becomes clear
409   * when looking at all of this through an object-oriented lens is
410   * that it is the {@link EventCache} (the {@code delta_fifo}, in the
411   * Go code) that is ultimately in charge of synchronizing.  It is
412   * not clear why this is a function of a reflector.  In an
413   * object-oriented world, perhaps the {@link EventCache} itself
414   * should be in charge of resynchronization schedules.</p>
415   *
416   * @return {@code true} if this {@link Reflector} should cause its
417   * {@link EventCache} to {@linkplain EventCache#synchronize()
418   * synchronize}; {@code false} otherwise
419   */
420  protected boolean shouldSynchronize() {
421    final String cn = this.getClass().getName();
422    final String mn = "shouldSynchronize";
423    if (this.logger.isLoggable(Level.FINER)) {
424      this.logger.entering(cn, mn);
425    }
426    final boolean returnValue = this.synchronizationExecutorService != null;
427    if (this.logger.isLoggable(Level.FINER)) {
428      this.logger.exiting(cn, mn, Boolean.valueOf(returnValue));
429    }
430    return returnValue;
431  }

441  /**
442   * Using the {@code operation} supplied at construction time,
443   * {@linkplain Listable#list() lists} appropriate Kubernetes
444   * resources, and then, on a separate {@link Thread}, {@linkplain
445   * VersionWatchable sets up a watch} on them, calling {@link
446   * EventCache#replace(Collection, Object)} and {@link
447   * EventCache#add(Object, AbstractEvent.Type, HasMetadata)} methods
448   * as appropriate.
449   *
450   * <p>The calling {@link Thread} is not blocked by invocations of
451   * this method.</p>
452   *
453   * @see #close()
454   */
455  @NonBlocking
456  public final void start() {
457    final String cn = this.getClass().getName();
458    final String mn = "start";
459    if (this.logger.isLoggable(Level.FINER)) {
460      this.logger.entering(cn, mn);
461    }
462
463    synchronized (this) {
464      if (this.watch == null) {
465        
466        // Run a list operation, and get the resourceVersion of that list.
467        @SuppressWarnings("unchecked")
468        final KubernetesResourceList<? extends T> list = ((Listable<? extends KubernetesResourceList<? extends T>>)this.operation).list();
469        assert list != null;
470
471        final ListMeta metadata = list.getMetadata();
472        assert metadata != null;
473        final String resourceVersion = metadata.getResourceVersion();
474        assert resourceVersion != null;
475        
476        // Using the results of that list operation, do a full replace
477        // on the EventCache with them.
478        final Collection<? extends T> replacementItems;
479        final Collection<? extends T> items = list.getItems();
480        if (items == null || items.isEmpty()) {
481          replacementItems = Collections.emptySet();
482        } else {
483          replacementItems = Collections.unmodifiableCollection(new ArrayList<>(items));
484        }
485        synchronized (eventCache) {
486          this.eventCache.replace(replacementItems, resourceVersion);
487        }
488        
489        // Record the resource version we captured during our list
490        // operation.
491        this.setLastSynchronizationResourceVersion(resourceVersion);
492        
493        // Now that we've vetted that our list operation works (i.e. no
494        // syntax errors, no connectivity problems) we can schedule
495        // resynchronizations if necessary.
496        this.setUpSynchronization();
497        
498        // Now that we've taken care of our list() operation, set up our
499        // watch() operation.
500        @SuppressWarnings("unchecked")
501        final Closeable temp = ((VersionWatchable<? extends Closeable, Watcher<T>>)operation).withResourceVersion(resourceVersion).watch(new WatchHandler());
502        assert temp != null;
503        this.watch = temp;
504        
505      }
506    }
507    
508    if (this.logger.isLoggable(Level.FINER)) {
509      this.logger.exiting(cn, mn);
510    }
511  }

563    /**
564     * Calls the {@link EventCache#add(Object, AbstractEvent.Type,
565     * HasMetadata)} method on the enclosing {@link Reflector}'s
566     * associated {@link EventCache} with information harvested from
567     * the supplied {@code resource}, and using an {@link Event.Type}
568     * selected appropriately given the supplied {@link
569     * Watcher.Action}.
570     *
571     * @param action the kind of Kubernetes event that happened; must
572     * not be {@code null}
573     *
574     * @param resource the {@link HasMetadata} object that was
575     * affected; must not be {@code null}
576     *
577     * @exception NullPointerException if {@code action} or {@code
578     * resource} was {@code null}
579     *
580     * @exception IllegalStateException if another error occurred
581     */
582    @Override
583    public final void eventReceived(final Watcher.Action action, final T resource) {
584      final String cn = this.getClass().getName();
585      final String mn = "eventReceived";
586      if (logger.isLoggable(Level.FINER)) {
587        logger.entering(cn, mn, new Object[] { action, resource });
588      }
589      Objects.requireNonNull(action);
590      Objects.requireNonNull(resource);
591
592      final ObjectMeta metadata = resource.getMetadata();
593      assert metadata != null;
594
595      final Event.Type eventType;
596
597      switch (action) {
598      case ADDED:
599        eventType = Event.Type.ADDITION;
600        break;
601      case MODIFIED:
602        eventType = Event.Type.MODIFICATION;
603        break;
604      case DELETED:
605        eventType = Event.Type.DELETION;
606        break;
607      case ERROR:        
608        // Uh...the Go code has:
609        //
610        //   if event.Type == watch.Error {
611        //     return apierrs.FromObject(event.Object)
612        //   }
613        //
614        // Now, apierrs.FromObject is here:
615        // https://github.com/kubernetes/apimachinery/blob/kubernetes-1.9.2/pkg/api/errors/errors.go#L80-L88
616        // This is looking for a Status object.  But
617        // WatchConnectionHandler will never forward on such a thing:
618        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L246-L258
619        //
620        // So it follows that if by some chance we get here, resource
621        // will definitely be a HasMetadata.  We go back to the Go
622        // code again, and remember that if the type is Error, the
623        // equivalent of this watch handler simply returns and goes home.
624        //
625        // Now, if we were to throw a RuntimeException here, which is
626        // the idiomatic equivalent of returning and going home, this
627        // would cause a watch reconnect:
628        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L159-L205
629        // ...up to the reconnect limit.
630        //
631        // ...which is fine, but I'm not sure that in an error case a
632        // WatchEvent will ever HAVE a HasMetadata as its payload.
633        // Which means MAYBE we'll never get here.  But if we do, all
634        // we can do is throw a RuntimeException...which ends up
635        // reducing to the same case as the default case below, so we
636        // fall through.
637      default:
638        eventType = null;
639        throw new IllegalStateException();
640      }
641
642      // Add an Event of the proper kind to our EventCache.
643      if (eventType != null) {
644        if (logger.isLoggable(Level.FINE)) {
645          logger.logp(Level.FINE, cn, mn, "Adding event to cache: {0} {1}", new Object[] { eventType, resource });
646        }
647        synchronized (eventCache) {
648          eventCache.add(Reflector.this, eventType, resource);
649        }
650      }
651
652      // Record the most recent resource version we're tracking to be
653      // that of this last successful watch() operation.  We set it
654      // earlier during a list() operation.
655      setLastSynchronizationResourceVersion(metadata.getResourceVersion());
656
657      if (logger.isLoggable(Level.FINER)) {
658        logger.exiting(cn, mn);
659      }
660    }

662    /**
663     * Invoked when the Kubernetes client connection closes.
664     *
665     * @param exception any {@link KubernetesClientException} that
666     * caused this closing to happen; may be {@code null}
667     */
668    @Override
669    public final void onClose(final KubernetesClientException exception) {
670      final String cn = this.getClass().getName();
671      final String mn = "onClose";
672      if (logger.isLoggable(Level.FINER)) {
673        logger.entering(cn, mn, exception);
674      }
675      if (exception != null && logger.isLoggable(Level.WARNING)) {
676        logger.logp(Level.WARNING, cn, mn, exception.getMessage(), exception);
677      }
678      if (logger.isLoggable(Level.FINER)) {
