280  @SuppressWarnings("rawtypes") // kubernetes-client's implementations of KubernetesResourceList use raw types
281  public <X extends Listable<? extends KubernetesResourceList> & VersionWatchable<? extends Closeable, Watcher<T>>> Reflector(final X operation,
282                                                                                                                              final EventCache<T> eventCache,
283                                                                                                                              final ScheduledExecutorService synchronizationExecutorService,
284                                                                                                                              final Duration synchronizationInterval,
285                                                                                                                              final Function<? super Throwable, Boolean> synchronizationErrorHandler) {
286    super();
287    this.logger = this.createLogger();
288    if (this.logger == null) {
289      throw new IllegalStateException("createLogger() == null");
290    }
291    final String cn = this.getClass().getName();
292    final String mn = "<init>";
293    if (this.logger.isLoggable(Level.FINER)) {
294      this.logger.entering(cn, mn, new Object[] { operation, eventCache, synchronizationExecutorService, synchronizationInterval });
295    }
296    Objects.requireNonNull(operation);
297    this.eventCache = Objects.requireNonNull(eventCache);
298    // TODO: research: maybe: operation.withField("metadata.resourceVersion", "0")?
299    this.operation = withResourceVersion(operation, "0");
300
301    if (synchronizationInterval == null) {
302      this.synchronizationIntervalInSeconds = 0L;
303    } else {
304      this.synchronizationIntervalInSeconds = synchronizationInterval.get(ChronoUnit.SECONDS);
305    }
306    if (this.synchronizationIntervalInSeconds <= 0L) {
307      this.synchronizationExecutorService = null;
308      this.shutdownSynchronizationExecutorServiceOnClose = false;
309      this.synchronizationErrorHandler = null;
310    } else {
311      this.synchronizationExecutorService = synchronizationExecutorService;
312      this.shutdownSynchronizationExecutorServiceOnClose = synchronizationExecutorService == null;
313      if (synchronizationErrorHandler == null) {
314        this.synchronizationErrorHandler = t -> {
315          if (this.logger.isLoggable(Level.SEVERE)) {
316            this.logger.logp(Level.SEVERE, this.getClass().getName(), "<synchronizationTask>", t.getMessage(), t);
317          }
318          return true;
319        };
320      } else {
321        this.synchronizationErrorHandler = synchronizationErrorHandler;
322      }
323    }
324    
325    if (this.logger.isLoggable(Level.FINER)) {
326      this.logger.exiting(cn, mn);
327    }
328  }

395  private synchronized final void closeSynchronizationExecutorService() {
396    final String cn = this.getClass().getName();
397    final String mn = "closeSynchronizationExecutorService";
398    if (this.logger.isLoggable(Level.FINER)) {
399      this.logger.entering(cn, mn);
400    }
401
402    this.cancelSynchronization();
403
404    if (this.synchronizationExecutorService != null && this.shutdownSynchronizationExecutorServiceOnClose) {
405
406      // Stop accepting new tasks.  Not that any will be showing up
407      // anyway, but it's the right thing to do.
408      this.synchronizationExecutorService.shutdown();
409
410      try {
411        if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
412          this.synchronizationExecutorService.shutdownNow();
413          if (!this.synchronizationExecutorService.awaitTermination(60L, TimeUnit.SECONDS)) {
414            if (this.logger.isLoggable(Level.WARNING)) {
415              this.logger.logp(Level.WARNING, cn, mn, "synchronizationExecutorService did not terminate cleanly after 60 seconds");
416            }
417          }
418        }
419      } catch (final InterruptedException interruptedException) {
420        this.synchronizationExecutorService.shutdownNow();
421        Thread.currentThread().interrupt();
422      }
423      
424    }
425
426    if (this.logger.isLoggable(Level.FINER)) {
427      this.logger.exiting(cn, mn);
428    }
429  }

431  /**
432   * As the name implies, sets up <em>synchronization</em>, which is
433   * the act of the downstream event cache telling its associated
434   * event listeners that there are items remaining to be processed,
435   * and returns a {@link Future} reprsenting the scheduled, repeating
436   * task.
437   *
438   * <p>This method schedules repeated invocations of the {@link
439   * #synchronize()} method.</p>
440   *
441   * <p>This method may return {@code null}.</p>
442   *
443   * @return a {@link Future} representing the scheduled repeating
444   * synchronization task, or {@code null} if no such task was
445   * scheduled
446   *
447   * @see #synchronize()
448   *
449   * @see EventCache#synchronize()
450   */
451  private synchronized final Future<?> setUpSynchronization() {
452    final String cn = this.getClass().getName();
453    final String mn = "setUpSynchronization";
454    if (this.logger.isLoggable(Level.FINER)) {
455      this.logger.entering(cn, mn);
456    }
457
458    if (this.synchronizationIntervalInSeconds > 0L) {
459      if (this.synchronizationExecutorService == null || this.synchronizationExecutorService.isTerminated()) {
460        this.synchronizationExecutorService = Executors.newScheduledThreadPool(1);
461        if (this.synchronizationExecutorService instanceof ScheduledThreadPoolExecutor) {
462          ((ScheduledThreadPoolExecutor)this.synchronizationExecutorService).setRemoveOnCancelPolicy(true);
463        }
464      }
465      if (this.synchronizationTask == null) {
466        if (this.logger.isLoggable(Level.INFO)) {
467          this.logger.logp(Level.INFO, cn, mn, "Scheduling downstream synchronization every {0} seconds", Long.valueOf(this.synchronizationIntervalInSeconds));
468        }
469        this.synchronizationTask = this.synchronizationExecutorService.scheduleWithFixedDelay(this::synchronize, 0L, this.synchronizationIntervalInSeconds, TimeUnit.SECONDS);
470      }
471      assert this.synchronizationExecutorService != null;
472      assert this.synchronizationTask != null;
473    }
474
475    if (this.logger.isLoggable(Level.FINER)) {
476      this.logger.exiting(cn, mn, this.synchronizationTask);
477    }
478    return this.synchronizationTask;
479  }

481  private final void synchronize() {
482    final String cn = this.getClass().getName();
483    final String mn = "synchronize";
484    if (this.logger.isLoggable(Level.FINER)) {
485      this.logger.entering(cn, mn);
486    }
487
488    if (this.shouldSynchronize()) {
489      if (this.logger.isLoggable(Level.FINE)) {
490        this.logger.logp(Level.FINE, cn, mn, "Synchronizing event cache with its downstream consumers");
491      }
492      Throwable throwable = null;
493      synchronized (this.eventCache) {
494        try {
495          this.eventCache.synchronize();
496        } catch (final Throwable e) {
497          assert e instanceof RuntimeException || e instanceof Error;
498          throwable = e;
499        }
500      }
501      if (throwable != null && !this.synchronizationErrorHandler.apply(throwable)) {
502        if (throwable instanceof RuntimeException) {
503          throw (RuntimeException)throwable;
504        } else if (throwable instanceof Error) {
505          throw (Error)throwable;
506        } else {
507          assert !(throwable instanceof Exception) : "Signature changed for EventCache#synchronize()";
508        }
509      }
510    }
511
512    if (this.logger.isLoggable(Level.FINER)) {
513      this.logger.exiting(cn, mn);
514    }
515  }

517  /**
518   * Returns whether, at any given moment, this {@link Reflector}
519   * should cause its {@link EventCache} to {@linkplain
520   * EventCache#synchronize() synchronize}.
521   *
522   * <p>The default implementation of this method returns {@code true}
523   * if this {@link Reflector} was constructed with an explicit
524   * synchronization interval or {@link ScheduledExecutorService} or
525   * both.</p>
526   *
527   * <h2>Design Notes</h2>
528   *
529   * <p>This code follows the Go code in the Kubernetes {@code
530   * client-go/tools/cache} package.  One thing that becomes clear
531   * when looking at all of this through an object-oriented lens is
532   * that it is the {@link EventCache} (the {@code delta_fifo}, in the
533   * Go code) that is ultimately in charge of synchronizing.  It is
534   * not clear why in the Go code this is a function of a reflector.
535   * In an object-oriented world, perhaps the {@link EventCache}
536   * itself should be in charge of resynchronization schedules, but we
537   * choose to follow the Go code's division of responsibilities
538   * here.</p>
539   *
540   * @return {@code true} if this {@link Reflector} should cause its
541   * {@link EventCache} to {@linkplain EventCache#synchronize()
542   * synchronize}; {@code false} otherwise
543   */
544  protected boolean shouldSynchronize() {
545    final String cn = this.getClass().getName();
546    final String mn = "shouldSynchronize";
547    if (this.logger.isLoggable(Level.FINER)) {
548      this.logger.entering(cn, mn);
549    }
550    final boolean returnValue;
551    synchronized (this) {
552      returnValue = this.synchronizationExecutorService != null;
553    }
554    if (this.logger.isLoggable(Level.FINER)) {
555      this.logger.exiting(cn, mn, Boolean.valueOf(returnValue));
556    }
557    return returnValue;
558  }

569  /**
570   * Using the {@code operation} supplied at construction time,
571   * {@linkplain Listable#list() lists} appropriate Kubernetes
572   * resources, and then, on a separate {@link Thread}, {@linkplain
573   * VersionWatchable sets up a watch} on them, calling {@link
574   * EventCache#replace(Collection, Object)} and {@link
575   * EventCache#add(Object, AbstractEvent.Type, HasMetadata)} methods
576   * as appropriate.
577   *
578   * <p>For convenience only, this method returns a {@link Future}
579   * representing any scheduled synchronization task created as a
580   * result of the user's having supplied a {@link Duration} at
581   * construction time.  The return value may be (and often is) safely
582   * ignored.  Invoking {@link Future#cancel(boolean)} on the returned
583   * {@link Future} will result in the scheduled synchronization task
584   * being cancelled irrevocably.</p>
585   *
586   * <p>This method may return {@code null}.</p>
587   *
588   * <p>The calling {@link Thread} is not blocked by invocations of
589   * this method.</p> 
590   *
591   * @return a {@link Future} representing a scheduled synchronization
592   * operation, or {@code null}
593   *
594   * @exception IOException if a watch has previously been established
595   * and could not be {@linkplain Watch#close() closed}
596   *
597   * @see #close()
598   */
599  @NonBlocking
600  public final Future<?> start() throws IOException {
601    final String cn = this.getClass().getName();
602    final String mn = "start";
603    if (this.logger.isLoggable(Level.FINER)) {
604      this.logger.entering(cn, mn);
605    }
606
607    Future<?> returnValue = null;
608    synchronized (this) {
609
610      try {
611      
612        // If somehow we got called while a watch already exists, then
613        // close the old watch (we'll replace it).  Note that,
614        // critically, the onClose() method of our watch handler sets
615        // this reference to null, so if the watch is in the process
616        // of being closed, this little block won't be executed.
617        if (this.watch != null) {
618          final Closeable watch = this.watch;
619          this.watch = null;
620          watch.close();
621        }
622        
623        // Run a list operation, and get the resourceVersion of that list.
624        @SuppressWarnings("unchecked")
625        final KubernetesResourceList<? extends T> list = ((Listable<? extends KubernetesResourceList<? extends T>>)this.operation).list();
626        assert list != null;
627
628        final ListMeta metadata = list.getMetadata();
629        assert metadata != null;
630
631        final String resourceVersion = metadata.getResourceVersion();
632        assert resourceVersion != null;
633        
634        // Using the results of that list operation, do a full replace
635        // on the EventCache with them.
636        final Collection<? extends T> replacementItems;
637        final Collection<? extends T> items = list.getItems();
638        if (items == null || items.isEmpty()) {
639          replacementItems = Collections.emptySet();
640        } else {
641          replacementItems = Collections.unmodifiableCollection(new ArrayList<>(items));
642        }
643        synchronized (this.eventCache) {
644          this.eventCache.replace(replacementItems, resourceVersion);
645        }
646        
647        // Record the resource version we captured during our list
648        // operation.
649        this.setLastSynchronizationResourceVersion(resourceVersion);
650        
651        // Now that we've vetted that our list operation works (i.e. no
652        // syntax errors, no connectivity problems) we can schedule
653        // synchronizations if necessary.
654        //
655        // A synchronization is an operation where, if allowed, our
656        // eventCache goes through its set of known objects and--for
657        // any that are not enqueued for further processing
658        // already--fires a synchronization event of type
659        // MODIFICATION.  This happens on a schedule, not in reaction
660        // to an event.  This allows its downstream processors a
661        // chance to try to bring system state in line with desired
662        // state, even if no events have occurred (kind of like a
663        // heartbeat).  See
664        // https://engineering.bitnami.com/articles/a-deep-dive-into-kubernetes-controllers.html#resyncperiod.
665        this.setUpSynchronization();
666        returnValue = this.synchronizationTask;
667        
668        // Now that we've taken care of our list() operation, set up our
669        // watch() operation.
670        @SuppressWarnings("unchecked")
671        final Versionable<? extends Watchable<? extends Closeable, Watcher<T>>> versionableOperation = (Versionable<? extends Watchable<? extends Closeable, Watcher<T>>>)this.operation;
672        this.watch = withResourceVersion(versionableOperation, resourceVersion).watch(new WatchHandler());
673
674      } catch (final IOException | RuntimeException exception) {
675        this.cancelSynchronization();
676        if (this.watch != null) {
677          try {
678            // TODO: haven't seen it, but reason hard about deadlock
679            // here; see
680            // WatchHandler#onClose(KubernetesClientException) which
681            // *can* call start() (this method) with the monitor.  I
682            // *think* we're in the clear here:
683            // onClose(KubernetesClientException) will only (re-)call
684            // start() if the supplied KubernetesClientException is
685            // non-null.  In this case, it should be, because this is
686            // an ordinary close() call.
687            this.watch.close();
688          } catch (final IOException | RuntimeException suppressMe) {
689            exception.addSuppressed(suppressMe);
690          }
691          this.watch = null;
692        }
693        throw exception;
694      }
695    }
696
697    if (this.logger.isLoggable(Level.FINER)) {
698      this.logger.exiting(cn, mn, returnValue);
699    }
700    return returnValue;
701  }

891  /**
892   * A {@link Watcher} of Kubernetes resources.
893   *
894   * @author <a href="https://about.me/lairdnelson"
895   * target="_parent">Laird Nelson</a>
896   *
897   * @see Watcher
898   */
899  private final class WatchHandler implements Watcher<T> {
900
901
902    /*
903     * Constructors.
904     */
905
906
907    /**
908     * Creates a new {@link WatchHandler}.
909     */
910    private WatchHandler() {
911      super();
912      final String cn = this.getClass().getName();
913      final String mn = "<init>";
914      if (logger.isLoggable(Level.FINER)) {
915        logger.entering(cn, mn);
916        logger.exiting(cn, mn);
917      }
918    }
919
920
921    /*
922     * Instance methods.
923     */
924
925
926    /**
927     * Calls the {@link EventCache#add(Object, AbstractEvent.Type,
928     * HasMetadata)} method on the enclosing {@link Reflector}'s
929     * associated {@link EventCache} with information harvested from
930     * the supplied {@code resource}, and using an {@link Event.Type}
931     * selected appropriately given the supplied {@link
932     * Watcher.Action}.
933     *
934     * @param action the kind of Kubernetes event that happened; must
935     * not be {@code null}
936     *
937     * @param resource the {@link HasMetadata} object that was
938     * affected; must not be {@code null}
939     *
940     * @exception NullPointerException if {@code action} or {@code
941     * resource} was {@code null}
942     *
943     * @exception IllegalStateException if another error occurred
944     */
945    @Override
946    public final void eventReceived(final Watcher.Action action, final T resource) {
947      final String cn = this.getClass().getName();
948      final String mn = "eventReceived";
949      if (logger.isLoggable(Level.FINER)) {
950        logger.entering(cn, mn, new Object[] { action, resource });
951      }
952      Objects.requireNonNull(action);
953      Objects.requireNonNull(resource);
954
955      final ObjectMeta metadata = resource.getMetadata();
956      assert metadata != null;
957
958      final Event.Type eventType;
959
960      switch (action) {
961      case ADDED:
962        eventType = Event.Type.ADDITION;
963        break;
964      case MODIFIED:
965        eventType = Event.Type.MODIFICATION;
966        break;
967      case DELETED:
968        eventType = Event.Type.DELETION;
969        break;
970      case ERROR:
971        // Uh...the Go code has:
972        //
973        //   if event.Type == watch.Error {
974        //     return apierrs.FromObject(event.Object)
975        //   }
976        //
977        // Now, apierrs.FromObject is here:
978        // https://github.com/kubernetes/apimachinery/blob/kubernetes-1.9.2/pkg/api/errors/errors.go#L80-L88
979        // This is looking for a Status object.  But
980        // WatchConnectionHandler will never forward on such a thing:
981        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L246-L258
982        //
983        // So it follows that if by some chance we get here, resource
984        // will definitely be a HasMetadata.  We go back to the Go
985        // code again, and remember that if the type is Error, the
986        // equivalent of this watch handler simply returns and goes home.
987        //
988        // Now, if we were to throw a RuntimeException here, which is
989        // the idiomatic equivalent of returning and going home, this
990        // would cause a watch reconnect:
991        // https://github.com/fabric8io/kubernetes-client/blob/v3.1.8/kubernetes-client/src/main/java/io/fabric8/kubernetes/client/dsl/internal/WatchConnectionManager.java#L159-L205
992        // ...up to the reconnect limit.
993        //
994        // ...which is fine, but I'm not sure that in an error case a
995        // WatchEvent will ever HAVE a HasMetadata as its payload.
996        // Which means MAYBE we'll never get here.  But if we do, all
997        // we can do is throw a RuntimeException...which ends up
998        // reducing to the same case as the default case below, so we
999        // fall through.
1000      default:
1001        eventType = null;
1002        throw new IllegalStateException();
1003      }
1004
1005      // Add an Event of the proper kind to our EventCache.
1006      if (eventType != null) {
1007        if (logger.isLoggable(Level.FINE)) {
1008          logger.logp(Level.FINE, cn, mn, "Adding event to cache: {0} {1}", new Object[] { eventType, resource });
1009        }
1010        synchronized (eventCache) {
1011          eventCache.add(Reflector.this, eventType, resource);
1012        }
1013      }
1014
1015      // Record the most recent resource version we're tracking to be
1016      // that of this last successful watch() operation.  We set it
1017      // earlier during a list() operation.
1018      setLastSynchronizationResourceVersion(metadata.getResourceVersion());
1019
1020      if (logger.isLoggable(Level.FINER)) {
1021        logger.exiting(cn, mn);
1022      }
1023    }
1024
1025    /**
1026     * Invoked when the Kubernetes client connection closes.
1027     *
1028     * @param exception any {@link KubernetesClientException} that
1029     * caused this closing to happen; may be {@code null}
1030     */
1031    @Override
1032    public final void onClose(final KubernetesClientException exception) {
1033      final String cn = this.getClass().getName();
1034      final String mn = "onClose";
1035      if (logger.isLoggable(Level.FINER)) {
1036        logger.entering(cn, mn, exception);
1037      }
1038
1039      synchronized (Reflector.this) {
1040        // No need to close it; we're being called because it's
1041        // closing.
1042        Reflector.this.watch = null;
1043      }
1044      
1045      if (exception != null) {
1046        if (logger.isLoggable(Level.WARNING)) {
1047          logger.logp(Level.WARNING, cn, mn, exception.getMessage(), exception);
1048        }
1049        // See
1050        // https://github.com/kubernetes/client-go/blob/5f85fe426e7aa3c1df401a7ae6c1ba837bd76be9/tools/cache/reflector.go#L204.
1051        try {
1052          Reflector.this.start();
1053        } catch (final IOException ioException) {
1054          if (logger.isLoggable(Level.SEVERE)) {
1055            logger.logp(Level.SEVERE, cn, mn, ioException.getMessage(), ioException);
1056          }
1057        }
1058      }
1059      if (logger.isLoggable(Level.FINER)) {
1060        logger.exiting(cn, mn, exception);
1061      }
1062    }
