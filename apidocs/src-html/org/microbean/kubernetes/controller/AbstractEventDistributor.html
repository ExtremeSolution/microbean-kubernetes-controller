017package org.microbean.kubernetes.controller;
018
019import java.util.HashMap;
020import java.util.Map;
021import java.util.Objects;
022
023import java.util.function.Consumer;
024
025import io.fabric8.kubernetes.api.model.HasMetadata;
026
027import net.jcip.annotations.GuardedBy;
028
029/**
030 * An abstract {@link ResourceTrackingEventQueueConsumer} whose
031 * implementations are particularly suitable for supplying to the
032 * {@link EventQueueCollection#start(Consumer)} method and can
033 * distribute {@link Event}s in some fashion.
034 *
035 * @author <a href="https://about.me/lairdnelson"
036 * target="_parent">Laird Nelson</a>
037 *
038 * @see #fireEvent(Event)
039 *
040 * @see EventQueue
041 *
042 * @see EventQueueCollection#start(Consumer)
043 */
044public abstract class AbstractEventDistributor<T extends HasMetadata> extends ResourceTrackingEventQueueConsumer<T> {
045
046
047  /*
048   * Constructors.
049   */
050
051
052  /**
053   * Creates a new {@link AbstractEventDistributor}.
054   *
055   * @param knownObjects a mutable {@link Map} of {@link HasMetadata}
056   * objects indexed by their keys (often a pairing of namespace and
057   * name); must not be {@code null}; <strong>will have its contents
058   * changed</strong> by this {@link AbstractEventDistributor}'s
059   * {@link #accept(EventQueue)} method; <strong>will be synchronized
060   * on</strong> by this {@link AbstractEventDistributor}'s {@link
061   * #accept(EventQueue)} method
062   *
063   * @exception NullPointerException if {@code knownObjects} is {@code
064   * null}
065   *
066   * @see #accept(EventQueue)
067   */
068  protected AbstractEventDistributor(final Map<Object, T> knownObjects) {
069    super(knownObjects);
070  }
071
072
073  /*
074   * Instance methods.
075   */
076  
077
078  /**
079   * Implements the {@link
080   * ResourceTrackingEventQueueConsumer#accept(Event.Type,
081   * HasMetadata, HasMetadata)} method by creating an {@link Event}
082   * and calling the {@link #fireEvent(Event)} method with it.
083   *
084   * @param eventType the {@link Type} of the new {@link Event}; must
085   * not be {@code null}
086   *
087   * @param priorResource a {@link HasMetadata} representing the
088   * <em>prior state</em> of the {@linkplain Event#getResource() Kubernetes
089   * resource the new <code>Event</code> will primarily concern}; may
090   * be&mdash;<strong>and often is</strong>&mdash;null
091   *
092   * @param resource a {@link HasMetadata} representing a Kubernetes
093   * resource; must not be {@code null}
094   *
095   * @exception NullPointerException if {@code eventType} or {@code
096   * resource} is {@code null}
097   */
098  @Override
099  protected void accept(final Event.Type eventType, final T priorResource, final T resource) {
100    Objects.requireNonNull(eventType);
101    this.fireEvent(new Event<T>(this, eventType, priorResource, resource));
102  }
103  
104  /**
105   * Distributes the supplied {@link Event} in some way.
106   *
107   * <p>Implementations of this method should be relatively fast as
108   * this method dictates the speed of {@link EventQueue}
109   * processing.</p>
110   *
111   * @param event the {@link Event} to distribute; must not be {@code null}
112   *
113   * @exception NullPointerException if {@code event} is {@code null}
114   */
115  protected abstract void fireEvent(final Event<T> event);
116  
117}
