017package org.microbean.kubernetes.controller;
018
019import java.util.HashMap;
020import java.util.Map;
021import java.util.Objects;
022
023import java.util.function.Consumer;
024
025import io.fabric8.kubernetes.api.model.HasMetadata;
026
027import net.jcip.annotations.GuardedBy;
028
029/**
030 * An abstract {@link Consumer} of {@link EventQueue}s, particularly
031 * suitable for supplying to the {@link
032 * EventQueueCollection#start(Consumer)} method, whose implementations
033 * can distribute {@link Event}s in some fashion.
034 *
035 * @author <a href="https://about.me/lairdnelson"
036 * target="_parent">Laird Nelson</a>
037 *
038 * @see EventQueue
039 *
040 * @see EventQueueCollection#start(Consumer)
041 */
042public abstract class AbstractEventDistributor<T extends HasMetadata> implements Consumer<EventQueue<T>> {
043
044
045  /*
046   * Instance fields.
047   */
048
049
050  /**
051   * A mutable {@link Map} of {@link HasMetadata} objects indexed by
052   * their keys (often a pairing of namespace and name).
053   *
054   * <p>This field is never {@code null}.</p>
055   *
056   * <p>The value of this field is {@linkplain
057   * #AbstractEventDistributor(Map) supplied at construction time} and
058   * is written to by the {@link #accept(EventQueue)} method.</p>
059   *
060   * <p>This {@link AbstractEventDistributor} implementation
061   * <strong>synchronizes on this field's value</strong> when mutating
062   * its contents.</p>
063   */
064  @GuardedBy("itself")
065  private final Map<Object, T> knownObjects;
066
067
068  /*
069   * Constructors.
070   */
071
072
073  /**
074   * Creates a new {@link AbstractEventDistributor}.
075   *
076   * @param knownObjects a mutable {@link Map} of {@link HasMetadata}
077   * objects indexed by their keys (often a pairing of namespace and
078   * name); must not be {@code null}; <strong>will have its contents
079   * changed</strong> by this {@link AbstractEventDistributor}'s
080   * {@link #accept(EventQueue)} method; <strong>will be synchronized
081   * on</strong> by this {@link AbstractEventDistributor}'s {@link
082   * #accept(EventQueue)} method
083   *
084   * @exception NullPointerException if {@code knownObjects} is {@code
085   * null}
086   *
087   * @see #accept(EventQueue)
088   */
089  protected AbstractEventDistributor(final Map<Object, T> knownObjects) {
090    super();
091    this.knownObjects = Objects.requireNonNull(knownObjects);
092  }
093
094
095  /*
096   * Instance methods.
097   */
098  
099
100  /**
101   * Reads all {@link Event}s from the supplied {@link EventQueue} in
102   * order and creates and {@linkplain #fireEvent(Event) distributes}
103   * new {@link Event}s for them that incorporate appropriate
104   * {@linkplain Event#getPriorResource() prior resource state}
105   * derived from the {@link Map} of resources supplied at {@linkplain
106   * #AbstractEventDistributor(Map) construction time}.
107   *
108   * @param eventQueue the {@link EventQueue} to read; may be {@code
109   * null} in which case no action will be taken; <strong>will be
110   * synchronized on</strong> by this method
111   *
112   * @exception EventQueueCollection.TransientException if a transient
113   * error was encountered that might be cleared if the supplied
114   * {@code eventQueue} were requeued and consumed again later
115   *
116   * @see #fireEvent(Event)
117   */
118  @Override
119  public final void accept(final EventQueue<T> eventQueue) {
120    if (eventQueue != null) {
121      synchronized (eventQueue) {
122        final Object key = eventQueue.getKey();
123        if (key == null) {
124          throw new IllegalStateException("eventQueue.getKey() == null; eventQueue: " + eventQueue);
125        }
126        for (final Event<T> event : eventQueue) {
127          if (event != null) {
128            assert key.equals(event.getKey());
129            final Event.Type eventType = event.getType();
130            assert eventType != null;
131            final T newResource = event.getResource();
132            if (event.getPriorResource() != null) {
133              // TODO: unexpected state, but should just be ignored or
134              // logged
135            }
136            final T priorResource;
137            final Event.Type newEventType;
138            synchronized (this.knownObjects) {
139              priorResource = this.knownObjects.get(key);
140              if (eventType.equals(Event.Type.DELETION)) {
141                this.knownObjects.remove(key);
142                newEventType = Event.Type.DELETION;
143              } else {
144                assert eventType.equals(Event.Type.ADDITION) || eventType.equals(Event.Type.MODIFICATION) || eventType.equals(Event.Type.SYNCHRONIZATION);
145                this.knownObjects.put(key, event.getResource());
146                if (priorResource == null) {
147                  newEventType = Event.Type.ADDITION;
148                } else if (eventType.equals(Event.Type.SYNCHRONIZATION)) {
149                  newEventType = Event.Type.SYNCHRONIZATION;
150                } else {
151                  newEventType = Event.Type.MODIFICATION;
152                }
153              }
154            }
155            this.fireEvent(new Event<>(this, newEventType, priorResource, newResource));
156          }
157        }
158      }
159    }
160  }
161
162  /**
163   * Distributes the supplied {@link Event} in some way.
164   *
165   * @param event the {@link Event} to distribute; must not be {@code null}
166   *
167   * @exception NullPointerException if {@code event} is {@code null}
168   */
169  protected abstract void fireEvent(final Event<T> event);
170  
171}
