017package org.microbean.kubernetes.controller;
018
019import java.util.Map;
020import java.util.Objects;
021
022import java.util.function.Consumer;
023
024import io.fabric8.kubernetes.api.model.HasMetadata;
025
026import net.jcip.annotations.GuardedBy;
027
028/**
029 * A {@link Consumer} of {@link EventQueue}s that tracks the
030 * Kubernetes resources they contain before allowing subclasses to
031 * process their individual {@link Event}s.
032 *
033 * @author <a href="https://about.me/lairdnelson"
034 * target="_parent">Laird Nelson</a>
035 *
036 * @see #accept(Event.Type, HasMetadata, HasMetadata)
037 */
038public abstract class ResourceTrackingEventQueueConsumer<T extends HasMetadata> implements Consumer<EventQueue<T>> {
039
040
041  /*
042   * Instance fields.
043   */
044
045  
046  /**
047   * A mutable {@link Map} of {@link HasMetadata} objects indexed by
048   * their keys (often a pairing of namespace and name).
049   *
050   * <p>This field may be {@code null} in which case no resource
051   * tracking will take place.</p>
052   *
053   * <p>The value of this field is {@linkplain
054   * #AbstractEventDistributor(Map) supplied at construction time} and
055   * is <strong>synchronized on</strong> and written to, if non-{@code
056   * null}, by the {@link #accept(EventQueue)} method.</p>
057   *
058   * <p>This class <strong>synchronizes on this field's
059   * value</strong>, if it is non-{@code null}, when mutating its
060   * contents.</p>
061   */
062  @GuardedBy("itself")
063  private final Map<Object, T> knownObjects;
064
065
066  /*
067   * Constructors.
068   */
069
070
071  /**
072   * Creates a new {@link ResourceTrackingEventQueueConsumer}.
073   *
074   * @param knownObjects a mutable {@link Map} of {@link HasMetadata}
075   * objects indexed by their keys (often a pairing of namespace and
076   * name); must not be {@code null}; <strong>will have its contents
077   * changed</strong> by this {@link
078   * ResourceTrackingEventQueueConsumer}'s {@link #accept(EventQueue)}
079   * method; <strong>will be synchronized on</strong> by this {@link
080   * ResourceTrackingEventQueueConsumer}'s {@link #accept(EventQueue)}
081   * method
082   *
083   * @see #accept(EventQueue)
084   */
085  protected ResourceTrackingEventQueueConsumer(final Map<Object, T> knownObjects) {
086    super();
087    this.knownObjects = knownObjects;
088  }
089
090  /**
091   * {@linkplain EvnetQueue#iterator() Loops through} all the {@link
092   * Event}s in the supplied {@link EventQueue}, keeping track of the
093   * {@link HasMetadata} it concerns along the way by
094   * <strong>synchronizing on</strong> and writing to the {@link Map}
095   * {@linkplain #ResourceTrackingEventQueueConsumer(Map) supplied at
096   * construction time}.
097   *
098   * <p>Individual {@link Event}s are forwarded on to the {@link
099   * #accept(Event.Type, HasMetadata, HasMetadata)} method.</p>
100   *
101   * @param eventQueue the {@link EventQueue} to process; may be
102   * {@code null} in which case no action will be taken
103   *
104   * @see #accept(Event.Type, HasMetadata, HasMetadata)
105   */
106  @Override
107  public final void accept(final EventQueue<T> eventQueue) {
108    if (eventQueue != null) {
109      synchronized (eventQueue) {
110        final Object key = eventQueue.getKey();
111        if (key == null) {
112          throw new IllegalStateException("eventQueue.getKey() == null; eventQueue: " + eventQueue);
113        }        
114        for (final Event<T> event : eventQueue) {
115          if (event != null) {
116            assert key.equals(event.getKey());            
117            final Event.Type eventType = event.getType();
118            assert eventType != null;            
119            final T newResource = event.getResource();
120            if (event.getPriorResource() != null) {
121              // TODO: unexpected state, but should just be ignored or
122              // logged
123            }
124            final T priorResource;
125            final Event.Type newEventType;
126            if (this.knownObjects == null) {
127              priorResource = null;
128              newEventType = eventType;
129            } else {
130              synchronized (this.knownObjects) {
131                priorResource = this.knownObjects.get(key);
132                if (eventType.equals(Event.Type.DELETION)) {
133                  this.knownObjects.remove(key);
134                  newEventType = Event.Type.DELETION;
135                } else {
136                  assert eventType.equals(Event.Type.ADDITION) || eventType.equals(Event.Type.MODIFICATION) || eventType.equals(Event.Type.SYNCHRONIZATION);
137                  this.knownObjects.put(key, newResource);
138                  if (priorResource == null) {
139                    newEventType = Event.Type.ADDITION;
140                  } else if (eventType.equals(Event.Type.SYNCHRONIZATION)) {
141                    newEventType = Event.Type.SYNCHRONIZATION;
142                  } else {
143                    newEventType = Event.Type.MODIFICATION;
144                  }
145                }
146              }
147            }
148            this.accept(newEventType, priorResource, newResource);
149          }
150        }
151      }
152    }
153  }
154
155  /**
156   * Called to process the contents of a given {@link Event} from the
157   * {@link EventQueue} supplied to the {@link #accept(EventQueue)}
158   * method, <strong>with that {@link EventQueue}'s monitor
159   * held</strong>.
160   *
161   * <p>Implementations of this method should be relatively fast as
162   * this method dictates the speed of {@link EventQueue}
163   * processing.</p>
164   *
165   * @param eventType the {@link Event.Type} describing the event
166   * encountered in the {@link EventQueue}; must not be {@code null}
167   *
168   * @param priorResource the <em>prior state</em> of the Kubernetes
169   * resource this logical event concerns; may be {@code null}
170   *
171   * @param resource the Kubernetes resource this logical event
172   * concerns; must not be {@code null}
173   *
174   * @exception NullPointerException if {@code eventType} or {@code
175   * resource} is {@code null}
176   *
177   * @see #accept(EventQueue)
178   *
179   * @see Event#Event(Object, Event.Type, HasMetadata, HasMetadata)
180   */
181  protected abstract void accept(final Event.Type eventType, final T priorResource, final T resource);
182  
183}
