017package org.microbean.kubernetes.controller;
018
019import java.util.Map;
020import java.util.Objects;
021
022import java.util.function.Consumer;
023
024import java.util.logging.Level;
025import java.util.logging.Logger;
026
027import io.fabric8.kubernetes.api.model.HasMetadata;
028
029import net.jcip.annotations.GuardedBy;
030
031/**
032 * A {@link Consumer} of {@link EventQueue}s that tracks the
033 * Kubernetes resources they contain before allowing subclasses to
034 * process their individual {@link Event}s.
035 *
036 * <p>Typically you would supply an implementation of this class to a
037 * {@link Controller}.</p>
038 *
039 * @author <a href="https://about.me/lairdnelson"
040 * target="_parent">Laird Nelson</a>
041 *
042 * @see #accept(Event.Type, HasMetadata, HasMetadata)
043 *
044 * @see Controller
045 */
046public abstract class ResourceTrackingEventQueueConsumer<T extends HasMetadata> implements Consumer<EventQueue<? extends T>> {
047
048
049  /*
050   * Instance fields.
051   */
052
053  
054  /**
055   * A mutable {@link Map} of {@link HasMetadata} objects indexed by
056   * their keys (often a pairing of namespace and name).
057   *
058   * <p>This field may be {@code null} in which case no resource
059   * tracking will take place.</p>
060   *
061   * <p>The value of this field is {@linkplain
062   * #AbstractEventDistributor(Map) supplied at construction time} and
063   * is <strong>synchronized on</strong> and written to, if non-{@code
064   * null}, by the {@link #accept(EventQueue)} method.</p>
065   *
066   * <p>This class <strong>synchronizes on this field's
067   * value</strong>, if it is non-{@code null}, when mutating its
068   * contents.</p>
069   */
070  @GuardedBy("itself")
071  private final Map<Object, T> knownObjects;
072
073  /**
074   * A {@link Logger} for use by this {@link
075   * ResourceTrackingEventQueueConsumer} implementation.
076   *
077   * <p>This field is never {@code null}.</p>
078   *
079   * @see #createLogger()
080   */
081  protected final Logger logger;
082
083  
084  /*
085   * Constructors.
086   */
087
088
089  /**
090   * Creates a new {@link ResourceTrackingEventQueueConsumer}.
091   *
092   * @param knownObjects a mutable {@link Map} of {@link HasMetadata}
093   * objects indexed by their keys (often a pairing of namespace and
094   * name); must not be {@code null}; <strong>will have its contents
095   * changed</strong> by this {@link
096   * ResourceTrackingEventQueueConsumer}'s {@link #accept(EventQueue)}
097   * method; <strong>will be synchronized on</strong> by this {@link
098   * ResourceTrackingEventQueueConsumer}'s {@link #accept(EventQueue)}
099   * method
100   *
101   * @see #accept(EventQueue)
102   */
103  protected ResourceTrackingEventQueueConsumer(final Map<Object, T> knownObjects) {
104    super();
105    this.logger = this.createLogger();
106    if (this.logger == null) {
107      throw new IllegalStateException("createLogger() == null");
108    }
109    final String cn = this.getClass().getName();
110    final String mn = "<init>";
111    if (this.logger.isLoggable(Level.FINER)) {
112      final String knownObjectsString;
113      if (knownObjects == null) {
114        knownObjectsString = null;
115      } else {
116        synchronized (knownObjects) {
117          knownObjectsString = knownObjects.toString();
118        }
119      }      
120      this.logger.entering(cn, mn, knownObjectsString);
121    }
122    this.knownObjects = knownObjects;
123    if (this.logger.isLoggable(Level.FINER)) {
124      this.logger.exiting(cn, mn);
125    }
126  }
127
128
129  /*
130   * Instance methods.
131   */
132
133
134  /**
135   * Returns a {@link Logger} for use with this {@link
136   * ResourceTrackingEventQueueConsumer}.
137   *
138   * <p>This method never returns {@code null}.</p>
139   *
140   * <p>Overrides of this method must not return {@code null}.</p>
141   *
142   * @return a non-{@code null} {@link Logger}
143   */
144  protected Logger createLogger() {
145    return Logger.getLogger(this.getClass().getName());
146  }
147
148  
149  /**
150   * {@linkplain EventQueue#iterator() Loops through} all the {@link
151   * Event}s in the supplied {@link EventQueue}, keeping track of the
152   * {@link HasMetadata} it concerns along the way by
153   * <strong>synchronizing on</strong> and writing to the {@link Map}
154   * {@linkplain #ResourceTrackingEventQueueConsumer(Map) supplied at
155   * construction time}.
156   *
157   * <p>Individual {@link Event}s are forwarded on to the {@link
158   * #accept(Event.Type, HasMetadata, HasMetadata)} method.</p>
159   *
160   * @param eventQueue the {@link EventQueue} to process; may be
161   * {@code null} in which case no action will be taken
162   *
163   * @see #accept(Event.Type, HasMetadata, HasMetadata)
164   */
165  @Override
166  public final void accept(final EventQueue<? extends T> eventQueue) {
167    final String cn = this.getClass().getName();
168    final String mn = "accept";
169    if (eventQueue == null) {
170      if (this.logger.isLoggable(Level.FINER)) {
171        this.logger.entering(cn, mn, null);
172      }
173    } else {
174      synchronized (eventQueue) {
175        if (this.logger.isLoggable(Level.FINER)) {
176          this.logger.entering(cn, mn, eventQueue);
177        }
178        final Object key = eventQueue.getKey();
179        if (key == null) {
180          throw new IllegalStateException("eventQueue.getKey() == null; eventQueue: " + eventQueue);
181        }        
182        for (final Event<? extends T> event : eventQueue) {
183          if (event != null) {
184            assert key.equals(event.getKey());            
185            final Event.Type eventType = event.getType();
186            assert eventType != null;            
187            final T newResource = event.getResource();
188            if (event.getPriorResource() != null) {
189              // TODO: unexpected state, but should just be ignored or
190              // logged
191            }
192            final T priorResource;
193            final Event.Type newEventType;
194            if (this.knownObjects == null) {
195              priorResource = null;
196              newEventType = eventType;
197            } else {
198              synchronized (this.knownObjects) {
199                priorResource = this.knownObjects.get(key);
200                if (eventType.equals(Event.Type.DELETION)) {
201                  this.knownObjects.remove(key);
202                  newEventType = Event.Type.DELETION;
203                } else {
204                  assert eventType.equals(Event.Type.ADDITION) || eventType.equals(Event.Type.MODIFICATION) || eventType.equals(Event.Type.SYNCHRONIZATION);
205                  this.knownObjects.put(key, newResource);
206                  if (priorResource == null) {
207                    newEventType = Event.Type.ADDITION;
208                  } else if (eventType.equals(Event.Type.SYNCHRONIZATION)) {
209                    newEventType = Event.Type.SYNCHRONIZATION;
210                  } else {
211                    newEventType = Event.Type.MODIFICATION;
212                  }
213                }
214              }
215            }
216            this.accept(newEventType, priorResource, newResource);
217          }
218        }
219      }
220    }
221    if (this.logger.isLoggable(Level.FINER)) {
222      this.logger.exiting(cn, mn);
223    }
224  }
225
226  /**
227   * Called to process the contents of a given {@link Event} from the
228   * {@link EventQueue} supplied to the {@link #accept(EventQueue)}
229   * method, <strong>with that {@link EventQueue}'s monitor
230   * held</strong>.
231   *
232   * <p>Implementations of this method should be relatively fast as
233   * this method dictates the speed of {@link EventQueue}
234   * processing.</p>
235   *
236   * @param eventType the {@link Event.Type} describing the event
237   * encountered in the {@link EventQueue}; must not be {@code null}
238   *
239   * @param priorResource the <em>prior state</em> of the Kubernetes
240   * resource this logical event concerns; may be {@code null}
241   *
242   * @param resource the Kubernetes resource this logical event
243   * concerns; must not be {@code null}
244   *
245   * @exception NullPointerException if {@code eventType} or {@code
246   * resource} is {@code null}
247   *
248   * @see #accept(EventQueue)
249   *
250   * @see Event#Event(Object, Event.Type, HasMetadata, HasMetadata)
251   */
252  protected abstract void accept(final Event.Type eventType, final T priorResource, final T resource);
253  
254}
